[docs]    def __init__(self, settings):
        """
        Initialize NoiseSource.

        :param settings: pyna settings
        :type settings: Dict[str, Any]
        :param subset:
        :type subset:
        """

        # Initialize pyNA object
        self.pyna = pyna(settings=settings)

        # Load trajectory data
        self.pyna.load_trajectory()

        # Load aircraft
        self.pyna.load_aircraft()

        # Initialize solution dictionaries
        self.spl = dict()
        self.oaspl = dict()



[docs]    def run(self, comps, subset=np.array([1])) -> None:
        """
        Compute the spectral and directional distribution of the source sound pressure level (SPL) and overall sound pressure level (OASPL).

        :param comps: list of components to run
        :type comps: list
        :param subset:
        :type subset:
        :return: None
        """

        # Set subset to NoiseSourceDistribution
        self.subset = subset

        # Iterate through the components
        for comp in comps:

            # Initialize noise source solution matrices
            self.spl[comp] = np.zeros((24, 17))
            self.oaspl[comp] = np.zeros(17)

            # Run cases for different theta angles
            self.pyna.data.theta = np.linspace(10, 170, 17)
            for j, theta_j in enumerate(self.pyna.data.theta):

                # Generate openmdao problem
                self.pyna.create_problem_source_distribution(comp=comp, subset=subset, theta=theta_j)

                # Run model
                self.pyna.problem.run_model()

                # Save results
                self.spl[comp][:, j] = self.pyna.problem.get_val('levels.spl')
                self.oaspl[comp][j] = self.pyna.problem.get_val('levels.oaspl')

        # Write output
        return None



[docs]    def load_validation_data(self, comps) -> (Dict, Dict):
        """
        Load validation data for noise source spectral and directional distributions.

        :param comps: list of components to run
        :type comps: list

        :return: (data_val, data_val_s)
        :rtype: (dict, dict)
        """

        # Initialize
        data_val = dict()
        data_val_s = dict()

        for comp in comps:
            if comp == 'core':
                data_val[comp] = pd.read_excel(self.pyna.settings['case_directory']+'/validation/Core Module Source.xlsx', sheet_name='Full').values
                data_val_s[comp] = pd.read_excel(self.pyna.settings['case_directory']+'/validation/Core Module Source.xlsx', sheet_name='Suppressed').values

            elif comp == 'jet_mixing':
                data_val[comp] = pd.read_excel(self.pyna.settings['case_directory']+'/validation/Jet Module Source.xlsx', sheet_name='Full').values
                data_val_s[comp] = pd.read_excel(self.pyna.settings['case_directory']+'/validation/Jet Module Source.xlsx', sheet_name='Suppressed').values

            elif comp == 'inlet_BB':
                data_val[comp] = pd.read_excel(self.pyna.settings['case_directory']+'/validation/Fan Module Source.xlsx', sheet_name='Full Inlet BB').values
                data_val_s[comp] = pd.read_excel(self.pyna.settings['case_directory']+'/validation/Fan Module Source.xlsx', sheet_name='Suppressed Inlet BB').values

            elif comp == 'discharge_BB':
                data_val[comp] = pd.read_excel(self.pyna.settings['case_directory']+'/validation/Fan Module Source.xlsx', sheet_name='Full Discharge BB').values
                data_val_s[comp] = pd.read_excel(self.pyna.settings['case_directory']+'/validation/Fan Module Source.xlsx', sheet_name='Suppressed Discharge BB').values

            elif comp == 'inlet_RS':
                data_val[comp] = pd.read_excel(self.pyna.settings['case_directory']+'/validation/Fan Module Source.xlsx',sheet_name='Full Inlet RS').values
                data_val_s[comp] = pd.read_excel(self.pyna.settings['case_directory']+'/validation/Fan Module Source.xlsx', sheet_name='Suppressed Inlet RS').values

            elif comp == 'discharge_RS':
                data_val[comp] = pd.read_excel(self.pyna.settings['case_directory']+'/validation/Fan Module Source.xlsx', sheet_name='Full Discharge RS').values
                data_val_s[comp] = pd.read_excel(self.pyna.settings['case_directory']+'/validation/Fan Module Source.xlsx', sheet_name='Suppressed Discharge RS').values

            elif comp == 'fan_inlet':
                data_val[comp] = pd.read_excel(self.pyna.settings['case_directory']+'/validation/Fan Module Source.xlsx', sheet_name='Fan Inlet Full').values
                data_val_s[comp] = pd.read_excel(self.pyna.settings['case_directory']+'/validation/Fan Module Source.xlsx', sheet_name='Fan Inlet Suppressed').values

            elif comp == 'fan_discharge':
                data_val[comp] = pd.read_excel(self.pyna.settings['case_directory']+'/validation/Fan Module Source.xlsx', sheet_name='Fan Discharge Full').values
                data_val_s[comp] = pd.read_excel(self.pyna.settings['case_directory']+'/validation/Fan Module Source.xlsx', sheet_name='Fan Discharge Suppressed').values

        # Write output
        return data_val, data_val_s



[docs]    def plot_spl_distribution(self, comps) -> None:
        """
        Plot spectral and directional distribution of noise source sound pressure level (SPL)

        :param comps: list of components to run
        :type comps: list

        :return: None
        """

        font_size = 20
        n_contour = 10

        # Spectral broadband plot
        fig, ax = plt.subplots(2, 3, figsize=(25, 10))

        # Load validation data
        if self.pyna.settings['validation']:
            _, data_val_lst = NoiseSourceDistribution.load_validation_data(self, comps)

        # Iterate over components
        col = -1
        row = 0
        for i, comp in enumerate(comps):
            if i == 3:
                row = 1
                col = 0
            else:
                col = col + 1

            # Create title for subplots
            if comp == 'fan_inlet':
                title = 'Fan inlet'
            elif comp == 'fan_discharge':
                title = 'Fan discharge'
            elif comp == 'core':
                title = 'Core'
            elif comp == 'jet_mixing':
                title = 'Jet mixing'
            elif comp == 'airframe':
                title = 'Airframe'
            else:
                raise ValueError('Invalid component specified')

            # Determine levels for contours
            lvl = np.round(np.linspace(np.min(self.spl[comp]), np.max(self.spl[comp]), n_contour))

            # Plot verification
            if self.pyna.settings['validation']:
                data_val = data_val_lst[comp][26*self.subset:26*(self.subset+1)-1, :]

                CS1 = ax[row, col].contour(self.pyna.data.theta, self.pyna.data.f, data_val[1:, 1:], levels=lvl, linestyles='dashed', linewidths=3)
                # ax[row, col].clabel(CS1, inline=1, fontsize=font_size)
                h2, _ = CS1.legend_elements()

            CS = ax[row, col].contour(self.pyna.data.theta, self.pyna.data.f, self.spl[comp], levels=lvl)
            ax[row, col].grid()
            ax[row, col].clabel(CS, levels=np.int64(lvl), inline=1, fontsize=font_size, fmt = r'%.f')
            ax[row, col].tick_params(axis='both', labelsize=font_size)
            ax[row, col].set_xlabel(r'$\theta [deg]$', fontsize=font_size)
            ax[row, col].set_ylabel('f [Hz]', fontsize=font_size)
            ax[row, col].set_yscale('log')
            # CB = fig.colorbar(CS, ax=ax[row, col], orientation="horizontal", pad=0.2)
            # CB.set_label('SPL [dB]', fontsize=font_size)
            # CB.ax.tick_params(labelsize=font_size)
            h1, _ = CS.legend_elements()
            ax[row, col].set_title(title, fontsize=font_size)

        if self.pyna.settings['validation']:
            ax[1, 1].legend([h1[0], h2[0]], ['pyNA', 'STCA Data'], fontsize=font_size, loc='center left', bbox_to_anchor=(1.25, 0.5))
        else:
            ax[1, 1].legend([h1[0]], ['pyNA'], fontsize=font_size, loc='center left', bbox_to_anchor=(1.25, 0.5))

        fig.delaxes(ax[1][2])
        plt.subplots_adjust(wspace=0.3, hspace=0.4)

        # Write output
        return None
